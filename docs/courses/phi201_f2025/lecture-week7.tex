\begin{frame}{Existential Introduction}
\begin{block}{Rule}
From a particular instance $Fa$, we may infer that something is $F$:
\[
\frac{Fa}{\exists x\,Fx}\;(\text{EI})
\]
\end{block}

\section{Existential elimination}

\begin{frame}{Idea}

  What can be derived from $\exists x\varphi (x)$?

  \bigskip Pick an arbitrary name $a$. If a general claim $\psi$ can
  be derived from an instance $\varphi (a)$, without making any
  additional assumptions about $a$, then $\psi$ follows from
  $\exists x\varphi (x)$.



\end{frame}

\begin{frame}{Existential Elimination}

    \begin{tabular}{>{\raggedleft\arraybackslash}p{2.5cm}
      >{\centering\arraybackslash}p{1.0cm} p{4cm}
      >{\raggedright\arraybackslash}p{3.5cm}}
      $\Gamma$ & (m) & $\exists x\varphi (x)$ \\[0.5em]
      n        & (n) & $\varphi (a)$ & A \\[0.5em]
      $\Delta$ & (o) & $\psi$  \\[0.5em]
      $\Gamma ,\Delta\backslash\{ n\}$ & (p) & $\psi$ & m,n,o EE \end{tabular}
      
  
\end{frame}

\begin{frame}{Existential Elimination}

  \begin{itemize}
  \item \textbf{Restriction:} The name $a$ must not occur in any
    dependencies of the existential premise, or of the derived
    conclusion, except for the instance $\varphi(a)$ itself.
  \item Dependencies are the union of the dependencies of the
    existential sentence and those of the derived conclusion, minus
    dependency on the instance. \end{itemize}

\end{frame}

\begin{frame}{Quantifier order matters}

  \Large $\forall x\exists y\,\varphi (x,y)$ follows from
  $\exists y\forall x\, \varphi (x,y)$. But not vice versa.


\end{frame}

\begin{frame}{Quantifier negation equivalences}

  $\neg \exists xFx$ is equivalent to $\forall x\neg Fx$

  \bigskip $\neg \forall xFx$ is equivalent to $\exists x\neg Fx$

  \bigskip ``Equivalent'' means mutually derivable

\end{frame}
