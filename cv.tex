      \step[fieldsource=author, 
            match=\regexp{(\w+\s+Halvorson|Halvorson,\s*\w+|H\.\s+Halvorson|Halvorson,\s*H\.)}, 
            replace={HH}]
      % Then strip "HH": sole-authored works become author-less; co-authored

    }
  }
